            final int price, final String graduationYear) {
        super();
        this.color = color;
        this.wheelDiameter = wheelDiameter;
        this.price = price;
        this.graduationYear = graduationYear;
    }

        return color;

        this.color = color;

        this.wheelDiameter = wheelDiameter;

        return price;

        this.price = price;
        }
